## Ansible Task - Install Nginx on Pathnex Server

- name: Install Nginx on Pathnex server
  host: all
  become: yes


  tasks:
     - name: Install Nginx
       yum:
          name: Nginx
          state: present

##Git Integration
## Jenkins Pipeline - Checkout Git

Pipeline {
    agent any
    stages{
        stage('checkout'){
            steps{
                git branch : "main", url: "https://github.com/Pathnex/sample-java-app.git"

            }    
        }
        stage('List Files'){
            steps{
                sh 'ls -la'
            }
        }
    }
}
