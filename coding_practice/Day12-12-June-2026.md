## kubernetes - Rolling Update Strategy

apiVersion: apps/v1
kind: Deployment
metadata:
  name: pathnex-rolling
spec:
  replicas: 3
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 1
   selector:
     matchLabels:
       app: roll
    template:
      metadata:
        labels:
          app: roll
       spec:
         containers:
           - name: web
             image: nginx

# Jenkins Pipeline - multi-branch

pipeline{
    agent any
    environment{
        INSTITUTE_NAME = "Pathnex"
    }
    stages{
        stage('Checkout'){
            steps{
                checkout scm
            }
        }
        stage('Build'){
            steps{
                sh 'mvn clean compile'
            }
        }
    }
}
