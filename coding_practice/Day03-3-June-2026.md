##Terraform - Security Group
provider "aws"{
    region = "us-east-1"
}

resource "aws_security_group" "my_sg"{
    name: "ssh_sg"
    description= "my sg for ssh"

    ingress {
        from_port = 22
        to_port = 22
        protocol = "tcp"
        cidr_blocks = ["0.0.0.0/0"]
    }
}

resource "aws_instance" "my_instance"{
    ami = "ami_id"
    instance_type = "r61.4xlarge"
    vpc_security_group_ids = [aws_security_group.my_sg.id]

    tags = {
        Name = "Pathnex_EC2"
    }
}

# NodeJS Build
## Jenkins Pipeline - NodeJS Build

pipeline {
    agent any
    tools{
        nodejs 'NodeJS-16'
    }
    stages{
        stage('checkout'){
            steps{
                git url : 'https://github.com/Pathnex/sample-node-app.git'
            }
        }
        stage('Install Dependencies'){
            steps{
                sh 'npm install'
            }
        }
        stage('Build'){
            steps{
                sh 'npm run build'
            }
        }
    }
}
