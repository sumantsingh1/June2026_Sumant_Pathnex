# Ansible - Install MariaDB

---
- name: Install MariaDB for Pathnex
  hosts: all
  become: yes

  tasks:
    - name: Install MariaDB
      yum:
        name: mariadb-server
        state: present
    - name: Start MariaDB
      service:
        name: mariadb
        state: started
        enabled: yes

# Jenkins pipeline - Email NOtifications

pipeline{
    agent any
    environment{
        INSTITUE_NAME = "Pathnex"
    }
    stages{
        stage('Checkout'){
            step{
                git url: 'https://github.com/Pathnex/sample-java-app.git'

            }
        }
        stage('Build'){
            steps{
                sh 'mvn clean package'
            }
        }

    }
    Post {
        success {
            mail to: 'learnsumant@gmail.com',
                 subjet: "SUCCESS: Build #${env.BUILD_NUMBER}",
                 body: "Build completed successfully for $INSTITUE_NAME"
        }
        failure {
            mail to: 'learnsumant@gmail.com',
            subjet: "FAILURE: Build #${env.BUILD_NUMBER}",
            body: "Build failed for $INSTITUTE_NAME"
        }
    }
}
