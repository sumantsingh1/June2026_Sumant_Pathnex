    cidr_block = "10.0.1.0/24"

    }

resource "aws_internet_gateway" "pathnex-gw"{
    vpc_id = aws_vpc.pathnex-vpc.id

    tags = {
        Name = "myPathnex-IG"
    }

}

resource "aws_route_table" "Pathnex-RT"{
    vpc_id = aws_vpc.pathnex-vpc.id

    route {
        cidr_block = "0.0.0.0/0"
        gateway_id = aws_internet_gateway.pathnex-gw.id
    }

    tags = {
        Name = "MyPathnex-RT"
    }
}

resource "aws_route_table_association" "Pathnex-RTA"{
    subnet_id = aws_subnet.pathnex-subnet.id
    route_table_id = aws_route_table.Pathnex-RT.id
}

# 2. Ansible - Install Nginx and Deploy HTML file

---
- name : Install Nginx and Deploy Html file
  hosts: yes
  become: yes

  tasks: 
    - name: check Nginx is installed
      apt: 
        name: nginx
        state: present
        update_cache: yes
    - name: Create HTML file
      copy:
        content: "<h1>Welcome to Pathnex Server!</h1>"
        dest: /var/www/html/index.html
        mode: '0644'
    - name: check Nginx service is running and enabled
      service:
        name: nginx
        state: started
        enabled: yes


# 3.Kubernetes - create Deployment + Service

# Deployment
apiVersion: apps/v1
name: Deployment
metadata: 
  name: nginx-deployment
spec:
  replicas: 2
  selector:
    matchlLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:latest
          ports:
          containerPort: 80

---

#service
apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  selector:
    app: nginx
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80
  type: LoadBalancer
