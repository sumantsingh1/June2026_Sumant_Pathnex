## Kubernetes - Node Afffinity

kind: pod
metadata:
  name: pathnex-affinity
spec:
  afffinity:
    nodeAffinity:
      requiredDuringScheduledIgnoredDuringExecution:
        nodeSelectorTerms:
          - matchExpressions:
            - key: disktype
              operator: In
              values:
                - ssd
    containers:
      - name: web
        image: nginx

# Rollback Simulation

## Jenkins Pipeline - Rollback on Failure

pipeline {
    agent any
    environent{
        INSTITUTE_NAME = "Pathnex"
    }
    stages {
        stage('Deploy') {
            steps {
                script {
                    try {
                        sh 'echo "Deploying $INSTITUTE_NAME application..."'
                        sh 'exit 1' // simulate failure
                    } catch (Exception e){
                        echo "Deployment failed. Rolling back $INSTITURE_NAME application"
                    }
                }
            }
        }
    }
}
